import React, { useState } from 'react'
import { BsFillArrowLeftCircleFill, BsFillArrowRightCircleFill } from "react-icons/bs";
import car1 from "../../Assets/car1.jpg"
import car2 from "../../Assets/car2.jpg"
import car3 from "../../Assets/car3.jpg"

const carImages = [
    { id: 1, src: car1 },
    { id: 2, src: car2 },
    { id: 3, src: car3 }
]

const Home = () => {
    const [currentImage, setCurrentImage] = useState(0)

    const handlePrevImage = () => {
        setCurrentImage(currentImage === 0 ? carImages.length - 1 : currentImage - 1)
    }

    const handleNextImage = () => {
        setCurrentImage((currentImage + 1) % carImages.length)
    }

    return (
        <div className="my-10">
            <div className='h-96 container bg-slate-500 flex items-center'>
                <div className='w-full h-full flex items-center '>
                    <img src={carImages[currentImage].src} alt={`Car ${carImages[currentImage].id}`} className='w-full h-full object-fit-cover' />
                    <div className='section absolute top-0 bottom-0 left-0 right-0 flex items-center justify-between'>
                        <div className='opacity-40 hover:opacity-100 cursor-pointer' onClick={handlePrevImage}>
                            <BsFillArrowLeftCircleFill className='h-auto w-10 md:w-16' />
                        </div>
                        <div className='opacity-40 hover:opacity-100 cursor-pointer' onClick={handleNextImage}>
                            <BsFillArrowRightCircleFill className='h-auto w-10 md:w-16' />
                        </div>
                    </div>
                </div>
            </div>
        </div>
    )
}

export default Home
